\classname{Dragon Disciple}\label{comm:prestige:dragondisciple}
\vspace*{-8pt}
\quot{"My grandmother was a dragon, and I bet you'd taste good with ketchup."}

\abox{Editor's Note}{Iaimeki did not include any flavor text for this prestige class, so feel free to imagine some, or write your own and send it to me.}

\ability{Prerequisites:}{}
\listprereq
	\itemability{Race:}{Any non-dragon}
	\itemability{Skills:}{Knowledge (arcana) 8 ranks}
	\itemability{Feats:}{Dragon Descendent}
	\itemability{Languages:}{Draconic}
\end{list}\vspace*{8pt}

\ability{Hit Die:}{d12}

\ability{Class Skills:}{The dragon disciple's class skills (and the key ability for each skill) are Concentration (Con), Craft (Int), Diplomacy (Cha), Escape Artist (Dex), Intimidate (Cha), Knowledge (all skills, taken individually) (Int), Listen (Wis), Profession (Wis), Search (Int), Sense Motive (Wis), Spot (Wis), and Use Magic Device (Cha).}\\

A dragon disciple also gains additional class skills based on the type of their draconic progenitor:
\begin{itemize}
\item{\textit{Black:} Hide (Dex), Move Silently (Dex), and Swim (Str).}
\item{\textit{Blue:} Bluff (Cha), Hide (Dex), and Spellcraft (Int).}
\item{\textit{Brass:} Bluff (Cha), Gather Information (Cha), and Survival (Wis).}
\item{\textit{Bronze:} Disguise (Cha), Survival (Wis), and Swim (Str).}
\item{\textit{Copper:} Bluff (Cha), Hide (Dex), and Jump (Str).}
\item{\textit{Gold:} Disguise (Cha), Heal (Wis), and Swim (Str).}
\item{\textit{Green:} Bluff (Cha), Hide (Dex), and Move Silently (Dex).}
\item{\textit{Red:} Appraise (Int), Bluff (Cha), and Jump (Str).}
\item{\textit{Silver:} Bluff (Cha), Disguise (Cha), and Jump (Str).}
\item{\textit{White:} Hide (Dex), Move Silently (Dex), and Swim (Str).}
\end{itemize}

\ability{Skills/Level:}{6 + Intelligence Bonus}

\smallskip\noindent All of the following are class features of the dragon disciple prestige class.

\ability{Weapon and Armor Proficiency:}{Dragon disciples gain no proficiency with any weapon or armor.}

\ability{Spellcasting:}{At 2nd, 5th, and 8th levels, a dragon disciple casts spells (including gaining any new spell slots and spell knowledge) as if she had also gained a level in sorcerer.}

\ability{Bonus Feat:}{At 1st, 4th, 7th, and 10th levels, a dragon disciple gains a bonus feat chosen from Affine Immunity, Affine Resilience, Draconic Breath, or Draconic Senses.  If she already has all four feats, she may instead gain any feat she meets the prerequisites for.}

\ability{Dragon Blood:}{Unless a dragon disciple is an outsider, plant, elemental, or undead, her type changes to dragon and she gains the augmented subtype.}

\ability{Draconic Scales (Ex):}{A dragon disciple gains a racial natural armor bonus equal to her class level.  If she already has a natural armor bonus, that bonus instead increases by her class level.  She also gains a +4 enhancement bonus to her natural armor.  As she takes more levels in dragon disciple, her skin becomes increasingly scaly and takes on the hue of her draconic ancestor.}

\ability{Draconic Strength (Ex):} {A dragon disciple gains a racial bonus to Strength equal to twice her class level.  She also gains a +4 enhancement bonus to Strength.}

\ability{Draconic Toughness (Ex):}{A dragon disciple gains bonus hit points equal to seven times her class level.}

\ability{Draconic Weapons (Ex):}{At 2nd level, a dragon disciple gains a bite as a primary natural weapon and two claws as secondary natural weapons.  They deal damage according to the chart below.}

\ability{Alternate Form (Su):}{Starting at 3rd level, a dragon disciple can assume any animal or humanoid form of Medium size or smaller as a standard action three times per day.  She can remain in her animal or humanoid form until she chooses to assume a new one or return to her natural form.}

\ability{Size Increase (Ex):}{At 3rd and 9th levels, a dragon disciple doubles in height, her weight increases eight times, and her size increases one category.  If she becomes Large, her reach becomes 10 feet; if she becomes Huge, her reach becomes 15 feet.}

\ability{Draconic Wings (Ex):} At 4th level, a dragon disciple gains wings, giving her a fly speed, maneuverability class, and wing slam attacks as secondary natural weapons with damage according to the chart below.

\ability{Draconic Movement (Ex):}{Beginning at 5th level, a dragon disciple gains increased movement, additional movement modes, or both.

\begin{itemize}
\item{\textit{Black:} Their land speed increases by 20 feet and they gain a swim speed of 60 feet.}
\item{\textit{Blue:} Their fly speed increases by 30 feet and they gain a burrow speed of 30 feet.}
\item{\textit{Brass:} Their land speed increases by 20 feet and they gain a burrow speed of 30 feet.}
\item{\textit{Bronze:} Their fly speed increases by 30 feet and they gain a swim speed of 60 feet.}
\item{\textit{Copper:} Their fly speed increases by 60 feet.}
\item{\textit{Gold:} Their land speed increases by 20 feet and they gain a swim speed of 60 feet.}
\item{\textit{Green:} Their fly speed increases by 30 feet and they gain a swim speed of 60 feet.}
\item{\textit{Red:} Their fly speed increases by 60 feet.}
\item{\textit{Silver:} Their fly speed increases by 60 feet.}
\item{\textit{White:} Their land speed increases by 20 feet and they gain a burrow speed of 30 feet.}
\end{itemize}}

\ability{Frightful Presence (Ex):} {At 6th level, a dragon disciple gains frightful presence.  It automatically affects creatures with fewer HD than her character level within a radius of 180 feet whenever she attacks, charges, flies overhead, or casts a spell.  A potentially affected creature that succeeds on a Will save (DC 14 + \half Character Level + her Cha modifier) remains immune to her frightful presence for 24 hours.  On a failure, creatures with 4 or fewer HD become panicked for 4d6 rounds and those with 5 or more HD become shaken for 4d6 rounds.  Other dragons ignore her frightful presence.}

\ability{Improved Draconic Toughness (Ex):}{Beginning at 6th level, a dragon disciple gains bonus hit points equal to seven times her Con bonus.  If her Con bonus changes, the hit points granted by this ability also change.}

\ability{Tail Slap (Ex):} {At 7th level, a dragon disciple grows a tail and gains a tail slap as a secondary natural weapons that deals \half her Strength bonus plus dice according to the chart below as damage.}

\ability{Draconic Powers (Sp):}{Starting at 8th level, a dragon disciple can use some spell-like abilities shared with her draconic ancestor. Her caster level is equal to her character level, and she gets a +4 racial bonus on the save DC.

\begin{itemize}
\item{\textit{Black:} 3/day--\spell{charm reptiles};  1/day--\spell{corrupt water}, \spell{insect plague}.}
\item{\textit{Blue:} 3/day--\spell{Create/destroy water}; 1/day--\spell{mirage arcana}, \spell{veil}.}
\item{\textit{Brass:} 1/day--\spell{Control weather}, \spell{control winds}, \spell{summon djinni}.}
\item{\textit{Bronze:} 3/day--\spell{Control water}; 1/day--\spell{control weather}.}
\item{\textit{Copper:} 1/day--\spell{move earth}, \spell{transmute rock to mud} or \spell{mud to rock}, \spell{wall of stone}.}
\item{\textit{Gold:} 3/day--\spell{Detect gems}; 1/day--\spell{foresight}, \spell{geas/quest}, \spell{luck bonus} (only affects the dragon disciple, lasts 1d3+18 hours), \spell{sunburst}.}
\item{\textit{Green:} 3/day--\spell{Dominate person}; 1/day--\spell{command plants}.}
\item{\textit{Red:} 1/day-- \spell{Discern location}, \spell{find the path}.}
\item{\textit{Silver:} 1/day--\spell{Control weather}, \spell{control winds}, \spell{reverse gravity}.}
\item{\textit{White:} 3/day--\spell{Freezing fog} (as MM white dragon); 1/day--\spell{control weather}.}
\end{itemize}}

\ability{Draconic Immortality (Su):} {Beginning at 9th level, a dragon disciple becomes immortal.  She no longer takes ability score penalties for aging, though bonuses still accrue, and cannot be magically aged; any penalties she may have already incurred, however, remain in place.}

\ability{Dragon Form (Su):}{Starting at 10th level, once per day as a standard action, a dragon disciple can assume the form of a true dragon.  While in this form, her size increases another category, she gains a +16 bonus to Str, 150 temporary hit points, a +4 bonus to her saves, and a +4 bonus on the save DCs of her dragon disciple spell-like abilities and abilities received from draconic Affinity feats.  While transformed, she can make crush and tail sweep attacks, dealing 4d6 and 2d6 plus \half times her Strength modifier damage. She may remain in dragon form for up to an hour, and return to her normal form at any time with a standard action.}
